src/
  ├─ main.tsx           # App entry point
  ├─ App.tsx             # Main application layout + routes
  ├─ App.css
  ├─ index.css
  ├─ components/
  │   ├─ admin/         # Publicly accessible pages
  │   │   ├─ Admin.tsx  # Calls each of the below components via tabs...
  │   │   ├─ AdminActions.tsx  
  │   │   ├─ AdminIndustryTypes.tsx  
  │   │   ├─ AdminLocalGroups.tsx  
  │   │   ├─ AdminUsers.tsx  
  │   ├─ experts/         # Publicly accessible pages
  │   │   ├─ Experts.tsx  
  │   ├─ public/         # Publicly accessible pages
  │   │   ├─ Welcome.tsx  
  │   │   ├─ PublicMenu.tsx  
  │   ├─ user/           # Auth + profile components
  │   │   ├─ Login.tsx 
  │   │   ├─ Register.tsx 
  │   │   ├─ EditProfile.tsx   # Parent component; orchestrates EditMember, EditExpert, and EditExpertise views
  │   │   ├─ EditMember.tsx 
  │   │   ├─ EditExpert.tsx 
  │   │   ├─ EditExpertise.tsx 
  │   ├─ members/        # Member-only features (protected)
  │   │   ├─ MemberMenu.tsx  
  │   │   ├─ Dashboard.tsx  
  │   ├─ shared/         # Reusable components (ProtectedRoute, Navbar)
  │   │   ├─ ProtectedRoute.tsx  # Route wrapper that restricts access to authenticated users only.
  │   └─ reports/        # Reporting / analytics components
  ├─ context/
  │   ├─ AuthContext.ts   # Manages authentication state, user info, and login/logout actions.
  │   ├─ RefreshContext.ts     # Handles automatic access token refresh and auth revalidation logic.
  ├─ data/
  │   ├─ demoExperts.json   # Demo data for the Experts.
  └─ utils/              # API hooks, constants, helpers
      ├─ constants.ts      # Centralized configuration values, e.g., API_BASE_URL, endpoints, tokens. 
                           # Future-proofed for switching to environment variables.
      ├─ customFetch.ts    # Low-level wrapper around fetch() that handles token headers, refresh logic, 
                           # and automatic redirect on auth failure.
                           # ⚠️ Never call this directly from a React component! 
                           #     Always use useApiRequest() to wrap your calls.
      ├─ useApiRequest.ts  # React hook for making API requests safely. 
                           # Wraps customFetch and manages error state for components. 
                           # Handles access/refresh tokens and navigation on auth failure.
                           # Components should call this hook instead of customFetch directly.
